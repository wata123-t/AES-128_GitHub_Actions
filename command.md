#Google Cloud 認証: ターミナルで 
gcloud auth application-default login

# 1. 初期化: ターミナルで以下を実行し、Google Cloud用のプラグインを読み込みます。
terraform init

# 2. 計画確認: 何が作成されるか確認します。
terraform plan

# 3. 反映: 実際にGoogle Cloud上に作成します。
terraform apply

//////////////////////////////////////////////////////////////////////////////
// ●1-1. シミュレーション実行（データの蓄積）
//////////////////////////////////////////////////////////////////////////////
>docker compose build

>docker compose up aes-verify

//////////////////////////////////////////////////////////////////////////////
// ●1-2. 分析実行
//////////////////////////////////////////////////////////////////////////////
>docker compose run --rm dbt run

//////////////////////////////////////////////////////////////////////////////
// ● 1-3. 合否判定
//////////////////////////////////////////////////////////////////////////////
>docker compose run --rm dbt test

-------------------------------------------------------------------------------
-- Docker 終了コマンド
-------------------------------------------------------------------------------
docker compose down
docker system prune -a
